async function doQuery(){

  const name = document.getElementById('name').value.trim();

  const cls  = document.getElementById('cls').value.trim();

  const phone= document.getElementById('phone').value.trim();


  if(API_URL.indexOf('http') !== 0){

    log('❌ 錯誤：你還沒有把 API_URL 改成自己的部署網址！\n\n目前 API_URL = ' + API_URL);

    return;

  }


  const url = `${API_URL}?phone=${encodeURIComponent(phone)}&name=${encodeURIComponent(name)}&cls=${encodeURIComponent(cls)}`;

  log('🔄 正在連線...\n\n請求網址：\n' + url);


  try{

    const res = await fetch(url);

    log('✅ 連線成功！\nHTTP 狀態：' + res.status + '\n\n正在讀取回應...');

    const text = await res.text();

    log('📦 伺服器回傳的原始內容：\n\n' + text + '\n\n──────────\n嘗試解析 JSON...');


    let json;

    try { json = JSON.parse(text); }

    catch(parseErr){

      log('❌ 回傳的不是 JSON（多半是權限頁面或登入頁）\n\n原始內容開頭：\n' + text.substring(0, 300) +

          '\n\n👉 這通常代表部署權限不是「任何人」，或網址用了 /dev 而非 /exec。');

      return;

    }


    if(json.success){

      log('🎉 查詢成功！後端與資料都正常！\n\n資料內容：\n' + JSON.stringify(json.data, null, 2) +

          '\n\n👉 後端沒問題，請換回正式版前端即可。');

    }else{

      log('⚠️ 連線正常，但後端回覆：「' + json.message + '」\n\n👉 若是「查無資料」，請檢查試算表的姓名/班別/電話是否完全一致。');

    }

  }catch(err){

    log('❌ 連線失敗（fetch error）：\n' + err.message +

        '\n\n可能原因：\n1) API_URL 網址打錯\n2) 部署權限不是「任何人」\n3) 尚未重新部署新版本\n4) 此頁面在沙箱預覽中（請存成檔案在瀏覽器開）');

  }

}
